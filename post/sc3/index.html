  3. gene filter 6% 和94%之间的保留

    def gene_filter(data,name):
    x_percent = 6
    tmp_lst = []
    total_num = len(name)
    for item in data:
        count = 0
        for num in item[1:]:
            if float(num) != 0:
                count += 1
        percent = count / total_num
        if 0.01 * x_percent < percent < (1 - 0.01 * x_percent):
            tmp_lst.append(item[1:])
    return tmp_lst

def log_transform(data,name):
    res = np.array(data)
    res = res.T
    result = []
    for item in zip(name, res):
        tmp_list = [item[0]]
        for i in item[1]:
            tmp_list.append(math.log(float(i) + 1, 2))
        result.append(tmp_list)
    return result
