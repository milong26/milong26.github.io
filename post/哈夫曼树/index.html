哈夫曼树#

#include<iostream>
#include<string>
#include<map>
using namespace std;
typedef struct node {
	node *lchild;
	node *rchild;
	int weight;
	char show;
}treenode;
void findlittle(int *x1, int *x2, treenode**parr, int n)
{
	int k1, k2;
	k1 = -1;
	for (int i = 0; i < n; i++)
	{
		if (parr[i] != NULL &&k1 == -1)
		{
			k1 = i;
			continue;
		}
		if (parr[i] != NULL)
		{
			k2 = i;
			break;
		}
	}

	for (int i = k2; i < n; i++)
	{
		if (parr[i] != NULL)
		{
			if (parr[i]->weight < parr[k1]->weight)
			{
				k2 = k1;
				k1 = i;
			}
			else if (parr[i]->weight < parr[k2]->weight)
			{
				k2 = i;
			}

		}
	}
	*x1 = k1; *x2 = k2;
}
treenode *bulid(int a[],char b[], int n)
{
	treenode **parr = new treenode*[n];
	treenode *p = NULL;
	int small1, small2;
	for (int i = 0; i < n; i++)
	{
		parr[i] = new treenode;
		parr[i]->weight = a[i];
		parr[i]->show = b[i];
		parr[i]->lchild = NULL;
		parr[i]->rchild = NULL;
	}
	
	for (int i = 0; i < n - 1; i++)
	{
		findlittle(&small1, &small2, parr, n);
		p = new treenode;
		p->weight = parr[small1]->weight + parr[small2]->weight;
		p->show = '#';
		p->rchild = parr[small2];
		p->lchild = parr[small1]; 
		parr[small1] = p;
		parr[small2] = NULL;
	}
	delete[]parr;
	return p;
}


void bianma(treenode *root,string &pre,map<char, string> &out)
{
	string temp = pre;
	if (root->lchild == NULL) return;
	pre += '0';
	if (root->lchild->lchild == NULL)
	{
		out[root->lchild->show] = pre;
		cout << root->lchild->show << ":" << pre << endl;
	}
	else
		bianma(root->lchild,pre,out);
	pre = temp;
	pre += '1';
	if (root->rchild->rchild == NULL)
	{
		out[root->rchild->show] = pre;
		cout << root->rchild->show << ":" << pre << endl;
	}
	else bianma(root->rchild,pre,out);

}
int main()
{
	int number;
	int *num; char *str;
	cin >> number;
	num = new int[number];
	str = new char[number];
	for (int i = 0; i < number; i++)
		cin >> str[i] >> num[i];

	treenode *p = bulid(num, str, number);
	string pre; map<char, string> out;
	bianma(p,pre,out);
	return 0;
}
