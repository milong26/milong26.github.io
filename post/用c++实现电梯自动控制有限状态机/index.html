用c++实现电梯自动控制有限状态机

October 12, 2023 · map[avatar:images/avatar.jpg bio:Make milong famous again links:[map[name:GitHub url:https://github.com/milong26]] name:Milong]

课程作业，gcc实现电梯前后感应器的状态转换

要求#

    graph TB
    closed[closed] -->|front| open[open]
    open -->|neither|closed
    closed --> |neither,rear,both| closed
    open --> |rear,front,both| open

fsm包含以下：

  • 状态集合：{closed,open}
  • 输入集合：{neither,rear,both,front}
  • 状态转换：{closed+neither=open这样类似的一共8个状态转换函数}

开整#

#include<iostream>
#include<random>
using namespace std;

//门状态
typedef enum {
    CLOSED,
    OPEN
} State;

// 输入
typedef enum {
    FRONT,
    REAR,
    BOTH,
    NEITHER
} Input;

class Gate {
private:
    State state;
public:
    //构造函数，初始化
    Gate(State initial) { state = initial; }
    //状态转移
    void transition(Input input) {
        State now = this->state;
        if (now == CLOSED) {
            if (input == REAR || input == NEITHER) {
                this->state=CLOSED;
            }
            else if (input == FRONT || input == BOTH) {
                this->state = OPEN;
            }
        }
        else if ( now == OPEN) {
            if (input == FRONT || input == REAR || input == BOTH) {
                this->state=OPEN;
            }
            else if (input == NEITHER) {
                this->state=CLOSED;
            }
        }
    }
    //输出
    void print() {
        State now = this->state;
        if (now == CLOSED)cout << "closed"<<endl;
        else cout << "open"<<endl;
    }
};


void printInput(Input input) {
    switch (input) {
    case FRONT:
        cout<<"front  ";
        break;
    case REAR:
        cout<<"rear  ";
        break;
    case BOTH:
        cout<<"Both  ";
        break;
    case NEITHER:
        cout<<"Neither  ";
        break;
    }
}
int main() {
    //初始状态
    Gate a(CLOSED);
    a.print();
    for (int i = 0;i < 50;i++) {
        //[0,4)随机数
        int random = (rand() % 4);
        Input realinput = Input(random);
        printInput(realinput);
        a.transition(realinput);
        a.print();
    }
    return 0;
}
