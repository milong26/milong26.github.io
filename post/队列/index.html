队列

December 7, 2019 · map[avatar:images/avatar.jpg bio:Make milong famous again links:[map[name:GitHub url:https://github.com/milong26]] name:Milong]

先入先出

front恒等于0，队列长度rear+1；队列空：rear=-1

静态，删除需要Θ(n)

删除：front++， Θ(1)

队列逐渐向数组尾部“移动”

队列空——front > rear

队尾到达数组尾时，即rear==MaxSize – 1时进行Add操作

front > 0：队列移动到数组头部（front<-0），再Add，Θ(n)

front==0，失败

最坏情况下仍为Θ(n)

最后一个元素之后是第一个元素

location(i)=(location(1)+i–1) % MaxSize

front——location(1) – 1

rear——最后一个元素在数组中的位置

插入、删除均为Θ(1)

缺陷：数组长度固定，但是也有其应用，例如固定最大值的线程池的描述等

队列空和队列满的区分：只允许最多MaxSize-1个元素

  • 队列空：front == rear
  • 队列满：front == (rear + 1) % MaxSize

代码

template<class T>
class Queue {// FIFO objects
   public:
      Queue(int MaxQueueSize = 10);
      ~Queue() {delete [] queue;}
      bool IsEmpty() const {return front == rear;}
      bool IsFull() const {     return (((rear + 1) % MaxSize == front) ? 1 : 0);}
      T First() const; // return front element
      T Last() const; // return last element
      Queue<T>& Add(const T& x);
      Queue<T>& Delete(T& x);
 private:
      int front;   //队首
      int rear;    // 队尾
      int MaxSize; // 队列容量
      T *queue;    // element array
};

//构造函数
template<class T>
Queue<T>::Queue(int MaxQueueSize)
{// Create an empty queue whose capacity
   MaxSize = MaxQueueSize + 1;//留出1个额外空间
   queue = new T[MaxSize];
   front = rear = 0;
}

//队首元素
template<class T>
T Queue<T>::First() const
{// Return first element of queue.  Throw
 // OutOfBounds exception if the queue is empty.
   if (IsEmpty()) throw OutOfBounds();
   return queue[(front + 1) % MaxSize];
}

//队尾元素
template<class T>
T Queue<T>::Last() const
{// Return last element of queue.  Throw
 // OutOfBounds exception if the queue is empty.
   if (IsEmpty()) throw OutOfBounds();
   return queue[rear];
}

//插入
template<class T>
Queue<T>& Queue<T>::Add(const T& x)
{// Add x to the rear of the queue.  Throw
 // NoMem exception if the queue is full.
   if (IsFull()) throw NoMem();
   rear = (rear + 1) % MaxSize;
   queue[rear] = x;
   return *this;
}

//删除
template<class T>
Queue<T>& Queue<T>::Delete(T& x)
{// Delete first element and put in x.  Throw
 // OutOfBounds exception if the queue is empty.
   if (IsEmpty()) throw OutOfBounds();
   front = (front + 1) % MaxSize;
   x = queue[front];//？
   return *this;
}

两种指针方向：

  • front->rear
  • rear->front

插入：Θ(1)

删除：front->rear Θ(1) ，rear->front Θ(n)

template<class T>
class LinkedQueue
{
// FIFO objects
   public:
      LinkedQueue() {front = rear = 0;}
      ~LinkedQueue(); // destructor
      bool IsEmpty() const
           {return ((front) ? false : true);}
      bool IsFull() const;
      T First() const; // return first element
      T Last() const; // return last element
 LinkedQueue<T>& Add(const T& x);
      LinkedQueue<T>& Delete(T& x);
   private:
      Node<T> *front;  // pointer to first node
      Node<T> *rear;   // pointer to last node
    //与数组队列相比，减少了两个数据成员：MaxSize和*queue
};

//析构
template<class T>
LinkedQueue<T>::~LinkedQueue()
{// Queue destructor.  Delete all nodes.
   Node<T> *next;
   while (front) {
      next = front->link; 
      delete front; 
      front = next;
      }
}

//判断满

template<class T>
T LinkedQueue<T>::First() const
{// Return first element of queue.  Throw
 // OutOfBounds exception if the queue is empty.
   if (IsEmpty()) throw OutOfBounds();
   return front->data;
}

<class T>
T LinkedQueue<T>::Last() const
{// Return last element of queue.  Throw
 // OutOfBounds exception if the queue is empty.
   if (IsEmpty()) throw OutOfBounds();
   return rear->data;
}

//插入
template<class T>
LinkedQueue<T>& LinkedQueue<T>::Add(const T& x)
{
   Node<T> *p = new Node<T>;
   p->data = x;
   p->link = 0;//新建1节点并赋值
   if (front) rear->link = p;  
   else front = p;             
   rear = p;
   return *this;
}

//删除
template<class T>
LinkedQueue<T>& LinkedQueue<T>::Delete(T& x)
{
   if (IsEmpty()) throw OutOfBounds();
   x = front->data; //备份当前队首节点值
   Node<T> *p = front;
   front = front->link; //确定新的队首
   delete p; //释放原队首的空间
   return *this;
}

for ( int i=1; i<=n; i++ ) {   //逐行处理
