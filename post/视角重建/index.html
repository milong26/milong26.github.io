img_new = cv2.imread('new.jpg')

# 使用 ORB 检测特征点

kp1, des1 = orb.detectAndCompute(img_old, None)
kp2, des2 = orb.detectAndCompute(img_new, None)

# 特征点匹配
bf = cv2.BFMatcher(cv2.NORM_HAMMING, crossCheck=True)
matches = bf.match(des1, des2)
matches = sorted(matches, key=lambda x: x.distance)

# 提取匹配点
src_pts = np.float32([kp2[m.trainIdx].pt for m in matches]).reshape(-1,1,2)  # 新位置
dst_pts = np.float32([kp1[m.queryIdx].pt for m in matches]).reshape(-1,1,2)  # 旧位置

# 计算单应性矩阵 H
H, mask = cv2.findHomography(src_pts, dst_pts, cv2.RANSAC)

# 可以保存 H，方便下次使用
np.save('homography.npy', H)

# 测试：将新位置图像映射到旧位置视角
h, w = img_old.shape[:2]

import pyrealsense2 as rs
import numpy as np
import cv2

# ====== 1. 加载之前计算好的单应性矩阵 ======
H = np.load("homography.npy")  # shape (3,3)

# ====== 2. 配置 Realsense 相机 ======
pipeline = rs.pipeline()
config = rs.config()
config.enable_stream(rs.stream.color, 640, 480, rs.format.bgr8, 30)  # 30 FPS

# 开启相机
pipeline.start(config)

try:
    while True:
        # 获取帧
        frames = pipeline.wait_for_frames()
        color_frame = frames.get_color_frame()
        if not color_frame:
            continue

        # 转换为 numpy array
        img = np.asanyarray(color_frame.get_data())

        # ====== 显示原始图像 ======
        cv2.imshow("Original", img)

        # ====== 应用单应性矩阵 ======
        h, w = img.shape[:2]
        warped = cv2.warpPerspective(img, H, (w, h))
        cv2.imshow("Warped", warped)

        # 按 q 键退出
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

finally:
    pipeline.stop()
    cv2.destroyAllWindows()
  1. 先不管视角复原的精确度的问题，黑边的产生原理？黑色区域其实是 二维透视变换 warp 后常见的现象
    1. 用单应性矩阵 H 的逆映射找到对应原图中的位置：$$ \begin{bmatrix} x \ y \ 1 \end{bmatrix} = H^{-1} \begin{bmatrix} x’ \ y’ \ 1 \end{bmatrix}$$
    2. 从原图中取对应位置的像素值（通常做双线性插值）。
    3. 黑色区域出现的原因：
    • warp 后的目标像素位置在原图对应位置 超出了原图范围。
      • 例如 H 会把图像向左、上、旋转或者拉伸。
      • 输出图像边缘部分的像素在原图中没有对应值。
    • OpenCV 默认填充这些“无对应值”的像素为 0，即黑色 [0,0,0][0,0,0][0,0,0]。
  2. 怎样模拟这个黑色边缘？遵循透视编换
    • 边缘形状
    • 四个角可能变形，形成梯形、三角形或不规则多边形。
    • 取决于 H 对图像的旋转、缩放、平移效果。
  • 分布
    • warp 后黑色区域在图像外围最多。
    • 如果 H 包含大旋转或平移，黑色区域可能出现在上、下、左、右边缘甚至斜角。
  • 和原图尺寸的关系
    • 如果 warp 输出尺寸固定 (w, h)，超出原图的部分必然黑色。
    • 如果输出尺寸很大，黑色区域更多；输出尺寸刚好包住映射区域，黑色区域最小。

saved_mask_debug_count = 0   # 用于计数

  

def random_wrap_mask_gpu(image: torch.Tensor,

                               max_offset_ratio=0.4,

                               debug_dir="./debug_mask",

                                save_limit=10) -> torch.Tensor:

    """

    image: H x W x C, torch.float32, CUDA tensor

    返回 masked_image: H x W x C, 彩色区域保持原样，黑边随机生成（梯形/透视效果）

    """

    global saved_mask_debug_count

    H, W, C = image.shape

    device = image.device

  

    # ================== 1. 定义原始四角 ==================

    corners = torch.tensor([[0,0],[W-1,0],[W-1,H-1],[0,H-1]], device=device, dtype=torch.float32)

  

    # ================== 2. 生成梯形/透视随机黑边 ==================

    # 上下左右随机偏移

    offsets_top = (torch.rand(2, device=device)*2 - 1) * W * max_offset_ratio

    offsets_bottom = (torch.rand(2, device=device)*2 - 1) * W * max_offset_ratio

    offsets_left = (torch.rand(2, device=device)*2 - 1) * H * max_offset_ratio

    offsets_right = (torch.rand(2, device=device)*2 - 1) * H * max_offset_ratio

  

    poly = torch.stack([

        corners[0] + torch.tensor([offsets_top[0], offsets_left[0]], device=device),

        corners[1] + torch.tensor([offsets_top[1], offsets_right[0]], device=device),

        corners[2] + torch.tensor([offsets_bottom[1], offsets_right[1]], device=device),

        corners[3] + torch.tensor([offsets_bottom[0], offsets_left[1]], device=device)

    ], dim=0)  # 4 x 2

  

    # ================== 3. 构建坐标网格 ==================

    yy, xx = torch.meshgrid(torch.arange(H, device=device), torch.arange(W, device=device), indexing='ij')

    coords = torch.stack([xx, yy], dim=-1).float()  # H x W x 2

  

    # ================== 4. 使用向量叉乘判断凸多边形内部 ==================

    def cross2d(a,b):

        return a[...,0]*b[...,1] - a[...,1]*b[...,0]

  

    v0 = poly[1] - poly[0]

    v1 = poly[2] - poly[1]

    v2 = poly[3] - poly[2]

    v3 = poly[0] - poly[3]

  

    mask = (

        (cross2d(v0, coords - poly[0]) >= 0) &

        (cross2d(v1, coords - poly[1]) >= 0) &

        (cross2d(v2, coords - poly[2]) >= 0) &

        (cross2d(v3, coords - poly[3]) >= 0)

    )

  

    # ================== 5. 应用 mask ==================

    masked_image = image.clone()

    masked_image[~mask] = 0.0

  

    # ================== 6. debug 保存原图和 mask ==================

    if saved_mask_debug_count < save_limit:

        import os

        from torchvision.transforms.functional import to_pil_image

  

        os.makedirs(debug_dir, exist_ok=True)

  

        mask_pixel_count = (~mask).sum().item()

        ratio = mask_pixel_count / (H * W)

        print(f"[DEBUG {saved_mask_debug_count}] Mask ratio={ratio:.3f}")

  

        # 转 CPU

        img_to_save = image.clamp(0,1).permute(2,0,1).cpu()

        masked_to_save = masked_image.clamp(0,1).permute(2,0,1).cpu()

  

        # 文件名带编号

        to_pil_image(img_to_save).save(

            os.path.join(debug_dir, f"original_{saved_mask_debug_count:02d}.png")

        )

        to_pil_image(masked_to_save).save(

            os.path.join(debug_dir, f"masked_{saved_mask_debug_count:02d}.png")

        )

        saved_mask_debug_count += 1

  

    return masked_image
