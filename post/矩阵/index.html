#include<iostream>
using namespace std;
class matrix {
public:
	matrix(int a, int b);//构造 看一下二维数组
	~matrix();//析构 
	matrix operator+(matrix &mat);//加法
	matrix operator=(matrix &mat);//等于,没写拷贝构造函数还没法用
	matrix(const matrix &mat);//拷贝构造函数
	friend ostream& operator<<(ostream&out, matrix &mat);//输出
	friend istream& operator>>(istream&in, matrix &mat);//输入
private:
	int rows;
	int cols;
	int** p;
};
matrix::matrix(int n, int m)//初始化
{
	rows = n;
	cols = m;
	p = new int*[rows];
	for (int i = 0; i < rows; i++)
		p[i] = new int[cols];
}
matrix::~matrix()//析构函数，主要是看删除二维数组
{
	for (int i = 0; i < rows; i++)
		delete p[i];
	delete p;
}
matrix::matrix(const matrix &mat)
{
	rows = mat.rows;
	cols = mat.cols;
	p = new int*[rows];
	for (int i = 0; i < rows; i++)
	{
		p[i] = new int[cols];
		for (int j = 0; j < cols; j++)
			p[i][j] = mat.p[i][j];
	}
}
matrix matrix::operator+(matrix &mat)
{
	matrix matsum(rows, cols);//定义一个新的
	for (int i = 0; i < rows; i++)
		for (int j = 0; j < cols; j++)
			matsum.p[i][j] = p[i][j] + mat.p[i][j];
	return matsum;
}
matrix matrix::operator=( matrix &mat)
{
	for (int i = 0; i < mat.rows; i++)
		for (int j = 0; j < mat.cols; j++)
			p[i][j] = mat.p[i][j];
	//cout << "ok" << endl;
	return (*this);
}
ostream& operator<<(ostream &out, matrix &mat)
{
	for (int i = 0; i < mat.rows; i++)
	{
		for (int j = 0; j < mat.cols; j++)
			out << mat.p[i][j]<<" ";
		out << endl;
	}
	return out;
}
istream& operator>>(istream&in, matrix &mat)
{
	for (int i = 0; i < mat.rows; i++)
		for (int j = 0; j < mat.cols; j++)
			in >> mat.p[i][j];
	return in;
}

{
	int n, m;

	matrix mat1(n, m),mat2(n,m),mat3(n,m);
	cin >> mat1;
	cout <<"mat1:"<<endl<<mat1<<endl;
	cin >> mat2;
	cout <<"mat2:"<<endl<<mat2 << endl;
	mat3=mat1;
	cout <<"mat3:"<<endl<<mat3 << endl;
}

  • 下对角线：i-j=1 ；元素数n-1；M(i, j)－a[i-2]

  • 主对角线：i=j ；元素数n；M(i, j)－a[n-1+i-1]

  • 上对角线i-j=-1；元素数n-1；M(i, j)－a[2n-1+i-1]
