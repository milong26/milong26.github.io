maven连接jdbc

April 17, 2021 · map[avatar:images/avatar.jpg bio:Make milong famous again links:[map[name:GitHub url:https://github.com/milong26]] name:Milong]

前提：配置好mysql、maven

  1. 加依赖并下载完

    pom.xml

    <!-- https://mvnrepository.com/artifact/mysql/mysql-connector-java -->
<dependency>
    <groupId>mysql</groupId>
    <artifactId>mysql-connector-java</artifactId>
    <version>8.0.16</version>
</dependency>

  2. 配置数据库连接

    resources/dbinfo.properties

    # 描述DB常用配置信息
driver=com.mysql.cj.jdbc.Driver
url= jdbc:mysql://127.0.0.1:3306/db_user?serverTimezone=UTC&useSSL=false
# 用户名和密码
usn=root
pwd=root

  3. java类

    utils/DBConfig.java

    package com.test.utils;
import java.io.IOException;
import java.util.Properties;

//获取数据库配置信息
public class DBConfig {
//    static properties对象实例
    private static Properties prop=new Properties();
    static {
        try {
            prop.load(DBConfig.class.getClassLoader().getResourceAsStream("dbinfo.properties"));
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
    public static String driver= prop.getProperty("driver");
    public static String url=prop.getProperty("url");
    public static String usn=prop.getProperty("usn");
    public static String pwd=prop.getProperty("pwd");
}

    dao/testJDBC.java中可以测试
