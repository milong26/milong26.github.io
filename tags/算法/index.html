哈夫曼树 #include<iostream> #include<string> #include<map> using namespace std; typedef struct node { node *lchild; node *rchild; int weight; char show; }treenode; void findlittle(int *x1, int *x2, treenode**parr, int n) { int k1, k2; k1 = -1; for (int i = 0; i < n; i++) { if (parr[i] != NULL &&k1 == -1) { k1 = i; continue; } if (parr[i] != NULL) { k2 = i; break; } } for (int i = k2; i < n; i++) { if (parr[i] != NULL) { if (parr[i]->weight < parr[k1]->weight) { k2 = k1; k1 = i; } else if (parr[i]->weight < parr[k2]->weight) { k2 = i; } } } *x1 = k1; *x2 = k2; } treenode *bulid(int a[],char b[], int n) { treenode **parr = new treenode*[n]; treenode *p = NULL; int small1, small2; for (int i = 0; i < n; i++) { parr[i] = new treenode; parr[i]->weight = a[i]; parr[i]->show = b[i]; parr[i]->lchild = NULL; parr[i]->rchild = NULL; } for (int i = 0; i < n - 1; i++) { findlittle(&small1, &small2, parr, n); p = new treenode; p->weight = parr[small1]->weight + parr[small2]->weight; p->show = '#'; p->rchild = parr[small2]; p->lchild = parr[small1]; parr[small1] = p; parr[small2] = NULL; } delete[]parr; return p; } void bianma(treenode *root,string &pre,map<char, string> &out) { string temp = pre; if (root->lchild == NULL) return; pre += '0'; if (root->lchild->lchild == NULL) { out[root->lchild->show] = pre; cout << root->lchild->show << ":" << pre << endl; } else bianma(root->lchild,pre,out); pre = temp; pre += '1'; if (root->rchild->rchild == NULL) { out[root->rchild->show] = pre; cout << root->rchild->show << ":" << pre << endl; } else bianma(root->rchild,pre,out); } int main() { int number; int *num; char *str; cin >> number; num = new int[number]; str = new char[number]; for (int i = 0; i < number; i++) cin >> str[i] >> num[i]; treenode *p = bulid(num, str, number); string pre; map<char, string> out; bianma(p,pre,out); return 0; }

求最短路径

题目描述 有N个城市，在每两个城市之间可以有一条交通道相连或者没有交通道连接. 现在有一些货物要从一座城市被转运到另一座城市，运输费由两部分组成： 1） 在途经的每两个城市之间的路上的运输花费。 2） 货物通过一座城市所要支付的某些税费，出发城市和目的地城市不必纳税。 请你编写一个程序，求出最小花费的路线。 输入： 在输入文件中，路上的花费、城市的税费、出发城市和目的地城市以如下的形式给出： 第一项输入：N，城市个数。程序序号下标从0开始。 第二项输入：Cost（N，N）的运输花费矩阵，花费为非负值，-1代表无路径。 第三项输入：Tax(N)每个城市的税费，非负值。 第四项输入：M，需要计算最小运输城市对的个数。 第五项输入：M对不超过N-1的整数，需要计算的最小花费的两个城市序号。 输出： 第一行输出路径，输出格式为i->j->m。下标序号加->。中间无空格。 第二行输出花费费用。 如果不存在路径，输出 No Path 测试数据： T1 输入: 0 3 22 -1 4 3 0 5 -1 -1 22 5 0 9 20 -1 -1 9 0 4 4 -1 20 4 0 5 17 8 3 1 3 0 2 2 4 1 3 输出: 21 2->3->4 16 1->0->4->3 17 T2 输入: 0 -1 6 0 10 22 1 0 1 输出: ...
